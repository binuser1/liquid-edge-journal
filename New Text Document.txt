    function renderGallery() {
      const filtered = mockDB.trades.filter(trade => {
        if (state.selectedFilter !== 'all' && trade.folder !== state.selectedFilter) return false;
        const tradeDate = new Date(trade.timestamp);
        if (tradeDate.getMonth() !== state.selectedMonth) return false;
        return true;
      });

      tradeGallery.innerHTML = '';
      if (filtered.length === 0) {
        tradeGallery.innerHTML = '<div style="grid-column:1/-1;text-align:center;padding:40px 20px;color:var(--text-secondary);">No trades recorded yet. Start trading! 🚀</div>';
        return;
      }

      filtered.forEach(trade => {
        const card = document.createElement('div');
        card.className = 'trade-card';
        const tradeDate = new Date(trade.timestamp);
        const dateStr = tradeDate.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
        const allChecksYes = trade.chk1 && trade.chk2 && trade.chk3 && trade.chk4;
        const statusClass = !allChecksYes && trade.notes ? 'danger' : (allChecksYes ? 'safe' : 'safe');
        const statusIcon = allChecksYes ? '✓' : (trade.guardianViolation ? '⚠' : '—');

        card.innerHTML = `
          <div class="card-chart">Chart</div>
          <div class="card-footer">
            <div class="card-market">${trade.market}</div>
            <div class="card-meta">
              <span class="card-date">${dateStr}</span>
              <span class="card-status ${statusClass}">${statusIcon}</span>
            </div>
          </div>
        `;
        card.addEventListener('click', (e) => {
          addRipple(e, card);
          openTradeModal(trade);
        });
        tradeGallery.appendChild(card);
      });
    }
    renderGallery();

    function deleteTradeFromModal(trade) {
      if (!confirm('Are you sure you want to delete this trade?')) return;

    function exportTradeData(trade) {
      const csv = `Trade Export\nDate,${new Date(trade.timestamp).toLocaleString()}\nMarket,${trade.market}\nEntry,${trade.entry || '—'}\nExit,${trade.exit || '—'}\nStop Loss,${trade.sl || '—'}\nTake Profit,${trade.tp || '—'}\nNotes,"${trade.notes}"\n`;
      const blob = new Blob([csv], { type: 'text/csv' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `trade_${new Date(trade.timestamp).getTime()}.csv`;
      a.click();
      URL.revokeObjectURL(url);
